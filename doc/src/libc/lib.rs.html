#[cfg(not(feature = "rustc-dep-of-std"))]
pub fn fwrite_slice<T: Sized>(slice: &[T], stream: *mut FILE) -> size_t {
    unsafe {
        fwrite(
            slice.as_ptr() as _,
            core::mem::size_of::<T>(),
            slice.len(),
            stream
        )
    }
